\begin{definition}[$m$-Cellular maps]
  A \textbf{CW skeleton} is an infinite sequence of types $$C_{-1} \xrightarrow{\iota_{-1}} C_0 \xrightarrow{\iota_{0}} C_1 \xrightarrow{\iota_{1}} \dots$$
  equipped with maps $\alpha : \sphere{n} \times A_n \to C_n$ where $A_n$ is equivalent to $\mathsf{Fin}(k_n)$ for some $k_n : \mathbb{N}$ and the following square is a pushout:
  \[
\begin{tikzcd}[ampersand replacement=\&]
	{\sphere{n} \times A_n} \& {A_n} \\
	{C_n} \& {C_{n+1}}
	\arrow[from=1-1, to=1-2]
	\arrow["{\alpha_n}"', from=1-1, to=2-1]
	\arrow["{\iota_n}"', from=2-1, to=2-2]
	\arrow[from=1-2, to=2-2]
	\arrow["\lrcorner"{anchor=center, pos=0.125, rotate=180}, draw=none, from=2-2, to=1-1]
\end{tikzcd}
\]
A \textbf{generalised CW skeleton} is a CW skeleton where each $A_n$ is only assumed to be merely finite.
\end{definition}
We write $\iota_\infty : C_n \to C_\infty$ for the (cell-wise) inclusion of $C_n$ into the colimit of $C_\bullet$.

\begin{definition}[Cellular $m$-maps]
  Given two CW skeleta $C_\bullet,D_\bullet$ an cellular $m$-map is a set of maps $f_n : C_n \to D_n$ for $n \leq m$ equipped with, for $n < m$, homotopies $h_n : \iota \circ f_{n} = f_{n+1} \circ \iota$
\end{definition}

\begin{definition}[Cellular $m$-homotopies]
  Given two CW skeleta $C_\bullet,D_\bullet$ and two cellular $m$-maps $f_n , g_n : C_n \to D_n$, an $m$-homotopy is ...
\end{definition}

\begin{theorem}[Cellular $m$-approximation theorem]
  Given two CW skeleta $C_\bullet,D_\bullet$, a map $f : C_\infty \to
  D_\infty$ and $m:\bN$, there exists (merely) an $m$-cellular map $f_\bullet : C_\bullet \to D_{\bullet}$ s.t. $\iota_{\infty} \circ f_m = f \circ \iota_{\infty}$.
\end{theorem}

\begin{theorem}[Cellular $m$-approximation theorem, part 2]
  Let $C_\bullet,D_\bullet$ be CW skeleta and consider two
  cellular $m$-maps map $f_n,g_n : C_n \to D_n$ with a
  homotopy $h_m : f_m = g_m$. In this case, there merely exists a
  cellular $m$-homotopy $f_\bullet \sim_m g_\bullet$
\end{theorem}

\begin{conjecture}\label{conj1}
  The approximation theorems hold even for $m = \infty$.
\end{conjecture}

\begin{conjecture}
  Conjecture~\ref{conj1} is unprovable for generalised CW skeleta.
\end{conjecture}
